# ProductDBMixedGrid

A grid component which handles screenshares, plugins and participants.

```jsx live
function CustomGrid() {
  const active = useProductDBSelector((m) => m.participants.active).toArray();
  const { meeting } = useProductDBMeeting();

  return (
    <ProductDBMixedGrid
      participants={active}
      pinnedParticipants={[]}
      screenShareParticipants={[meeting.self]}
      plugins={[]}
      meeting={meeting}
      style={{ height: '400px', width: '100%' }}
    />
  );
}
```
