ProductDB provides a default pre-call preview UI, also known as setup screen as part of our UI components.

Previously in the Quickstart example, we used the following component.

```tsx
<ProductDBMeeting meeting={meeting} showSetupScreen={true} />
```

If you want to break down the above for a custom UI but still want to reuse the default setup screen, use the following component.

import CodeRunner from '@site/src/components/CodeRunner/CodeRunner';

<CodeRunner 
highlight={[{start: 11, end: 11}]}
hide={[{start:1, end: 3}]}
file={`
import { ProductDBSetupScreen } from '@ProductDBsdk/react-ui-kit';
import { useProductDBMeeting, useProductDBSelector } from '@ProductDBsdk/react-web-core';

export default function MyMeeting() {
  const { meeting } = useProductDBMeeting();
  const roomState = useProductDBSelector((m) => m.self.roomState);

  return (
    <div className="flex w-full h-full">
      {roomState === 'init' && <ProductDBSetupScreen meeting={meeting} />}
      {roomState === 'joined' && <center>Custom in-meeting UI</center>}
      {roomState === 'ended' && <center>Custom post-meeting UI</center>}
    </div>
  );
}`
} />
