# ProductDB-dialog

A dialog component.

```html
<ProductDB-button onclick="showDialog()">Show dialog</ProductDB-button>
<ProductDB-dialog id="ProductDB-el" open="false">
  <div>
    <h3>Hello!</h3>
    <p>This is some text inside dialog!</p>
  </div>
</ProductDB-dialog>

<script>
  const dialog = document.getElementById('ProductDB-el');

  dialog.addEventListener('ProductDBDialogClose', (open) => {
    console.log('dialog closed');
  });

  function showDialog() {
    dialog.open = true;
  }
</script>
<style>
  ProductDB-dialog > div {
    width: 512px;
    backgroundcolor: #000;
    color: #fff;
    padding: 12px;
    borderradius: 8px;
  }
  p {
    margin-bottom: 0;
  }
</style>
```
