# ProductDB-poll

A poll component.

Shows a poll where a user can vote.

```html
<ProductDB-poll id="ProductDB-el"></ProductDB-poll>

<script>
  const el = document.getElementById('ProductDB-el');

  el.addEventListener('ProductDBVotePoll', (e) => {
    console.log('Voted', e.detail);
  });

  el.poll = {
    id: 'poll-id',
    question: 'Have you started using ProductDB yet?',
    options: [
      {
        text: 'Yes',
        votes: [{ id: 'vaibhavs-user-id', name: 'Vaibhav' }],
        count: 0,
      },
      {
        text: 'Nope',
        votes: [],
        count: 0,
      },
    ],
    anonymous: false,
    hideVotes: false,
    createdBy: 'Vaibhav',
    createdByUserId: 'vaibhavs-user-id',
    voted: [],
  };
</script>
```
