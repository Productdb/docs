# productdb-dialog-manager

A component which handles all dialog elements in a component such as:

- productdb-settings
- productdb-leave-meeting
- productdb-permissions-message
- productdb-image-viewer
- productdb-remote-access-manager
- productdb-breakout-rooms-manager

This components depends on the values from `states` object.

```html
<productdb-button onclick="showSettings()">Show Settings</productdb-button>
<productdb-button onclick="showLeaveConfirmation()">Show Settings</productdb-button>

<productdb-dialog-manager id="productdb-el"></productdb-dialog-manager>

<script>
  const dialog = document.getElementById('productdb-el');
  dialog.meeting = meeting;
  let states = {};

  function updateStates() {
    dialog.states = states;
  }

  function stateUpdate(s) {
    states = { ...states, ...s };
    updateStates();
  }

  function showSettings() {
    stateUpdate({ activeSettings: true });
  }

  function showLeaveConfirmation() {
    stateUpdate({ activeLeaveConfirmation: true });
  }

  dialog.addEventListener('dyteStateUpdate', (e) => {
    stateUpdate(e.detail);
  });

  updateStates();
</script>
```
