# productdb-dialog

A dialog component.

```html
<productdb-button onclick="showDialog()">Show dialog</productdb-button>
<productdb-dialog id="productdb-el" open="false">
  <div>
    <h3>Hello!</h3>
    <p>This is some text inside dialog!</p>
  </div>
</productdb-dialog>

<script>
  const dialog = document.getElementById('productdb-el');

  dialog.addEventListener('dyteDialogClose', (open) => {
    console.log('dialog closed');
  });

  function showDialog() {
    dialog.open = true;
  }
</script>
<style>
  productdb-dialog > div {
    width: 512px;
    backgroundcolor: #000;
    color: #fff;
    padding: 12px;
    borderradius: 8px;
  }
  p {
    margin-bottom: 0;
  }
</style>
```
